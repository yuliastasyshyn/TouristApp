-- Таблиця для ресторанів
CREATE TABLE Landmarks (
    id SERIAL PRIMARY KEY,               
    name VARCHAR(255) NOT NULL,          
    description TEXT,                  
    link VARCHAR(255),                   
    image_url VARCHAR(255)              
    --location VARCHAR(255) NOT NULL,      
    
);

-- Таблиця для магазинів
CREATE TABLE Shops (
    id SERIAL PRIMARY KEY,              
    name VARCHAR(255) NOT NULL,          
    description TEXT,                   
    link VARCHAR(255),                  
    image_url VARCHAR(255)             
   -- location VARCHAR(255) NOT NULL,      
    
);
